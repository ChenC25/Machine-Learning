In[101]:
# Label Encoding
le = LabelEncoder()
label_mappings = {}

for col in ['VRSELLTYPE', 'Vehicle_trantype', 'Vehicle_drive', 'Misc_SalesChannel','Vehicle_condition_drivable',
            'Vehicle_condition_crstatus',"Vehicle_airbag"]:
    cars_categorical[col] = le.fit_transform(cars_categorical[col])
    label_mappings[col] = dict(zip(le.classes_, le.transform(le.classes_)))

# Frequency Encoding
frequency_mappings = {}

for col in ['Make', 'Model', 'Vehicle_botcolor', 'VRLOCATION', 'Vehicle_btext']:
    freq_map = cars_categorical[col].value_counts().to_dict()
    cars_categorical[col] = cars_categorical[col].map(freq_map)
    frequency_mappings[col] = freq_map

# Handling Missing Values
cars_categorical['Vehicle_condition_crstatus'].fillna('Unknown', inplace=True)
cars_categorical['Vehicle_condition_drivable'].fillna('Unknown', inplace=True)
In[102]:
for col, mapping in label_mappings.items():
    cars_test_categorical[col] = cars_test_categorical[col].fillna('Unknown').map(mapping).fillna(-1).astype(int)
for col, freq_map in frequency_mappings.items():
    cars_test_categorical[col] = cars_test_categorical[col].map(freq_map).fillna(0).astype(int)

cars_test_categorical['Vehicle_condition_crstatus'] = cars_test_categorical['Vehicle_condition_crstatus'].fillna('Unknown')
cars_test_categorical['Vehicle_condition_drivable'] = cars_test_categorical['Vehicle_condition_drivable'].fillna('Unknown')

# Create the StackingRegressor
stacked_model = StackingRegressor(
    estimators=[('rf', rf), ('xgb', xgb)],
    final_estimator=XGBRegressor(
        n_estimators=50,
        learning_rate=0.1,
        random_state=42
    )
)
